Python Fundamentals with ChatGPT

Variables and Data Types

Python automatically determines data types:

# Strings
name = "Alice"
print(name)  # Output: Alice

# Numbers
age = 25
height = 5.8

# Booleans
is_student = True

ChatGPT Tip: Ask "What's the difference between int and float in Python?" for detailed explanations.

Basic Operations

# Arithmetic
a = 10
b = 3
print(a + b)   # 13
print(a - b)   # 7
print(a * b)   # 30
print(a / b)   # 3.333...
print(a // b)  # 3 (integer division)
print(a % b)   # 1 (remainder)
print(a ** b)  # 1000 (power)

String Manipulation

# Concatenation
first_name = "John"
last_name = "Doe"
full_name = first_name + " " + last_name
print(full_name)  # John Doe

# String methods
message = "Hello, Python!"
print(message.lower())      # hello, python!
print(message.upper())      # HELLO, PYTHON!
print(message.replace("Python", "ChatGPT"))  # Hello, ChatGPT!

Getting User Input

# Simple input
name = input("What's your name? ")
print(f"Hello, {name}!")

# Converting input to number
age = int(input("How old are you? "))
print(f"Next year you'll be {age + 1}")

Control Flow and Logic

If-Else Statements

age = 15

if age >= 18:
    print("You are an adult")
elif age >= 13:
    print("You are a teenager")
else:
    print("You are a child")

Comparison Operators

a = 10
b = 5

print(a == b)  # False
print(a != b)  # True
print(a > b)   # True
print(a < b)   # False
print(a >= b)  # True
print(a <= b)  # False

Logical Operators

age = 25
has_license = True

if age >= 18 and has_license:
    print("You can drive")

# OR example
if age < 13 or age > 70:
    print("Special consideration needed")

# NOT example
if not has_license:
    print("You need a license")

Loops - For Loop

# Loop through a range
for i in range(5):
    print(i)  # 0, 1, 2, 3, 4

# Loop through a list
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(fruit)

Loops - While Loop

count = 0
while count < 5:
    print(count)
    count += 1  # Same as count = count + 1

Working with Functions

Creating Your First Function

def greet():
    print("Hello!")

greet()  # Call the function

Functions with Parameters

def greet(name):
    print(f"Hello, {name}!")

greet("Alice")  # Hello, Alice!
greet("Bob")    # Hello, Bob!

Functions with Return Values

def add(a, b):
    return a + b

result = add(5, 3)
print(result)  # 8

Multiple Parameters and Default Values

def describe_person(name, age=18):
    print(f"{name} is {age} years old")

describe_person("Alice")        # Alice is 18 years old
describe_person("Bob", 25)      # Bob is 25 years old

Best Practices for Functions

  1. Use clear names: calculate_total() not calc()
  2. One task per function: Don't do everything in one function
  3. Document your code: Use comments
  4. Test your functions: Make sure they work correctly

Data Structures and Collections

Lists

# Creating a list
fruits = ["apple", "banana", "cherry"]

# Accessing elements
print(fruits[0])      # apple
print(fruits[-1])     # cherry (last item)

# Modifying lists
fruits.append("date")           # Add at end
fruits.insert(1, "blueberry")   # Insert at position
fruits.remove("banana")         # Remove specific item

# List operations
print(len(fruits))    # Number of items
print("apple" in fruits)  # Check if item exists

Dictionaries

# Creating a dictionary
person = {
    "name": "Alice",
    "age": 25,
    "city": "New York"
}

# Accessing values
print(person["name"])  # Alice

# Adding new key-value pair
person["email"] = "alice@example.com"

# Removing items
del person["city"]

Tuples

# Tuples are immutable (can't change)
coordinates = (10, 20)
print(coordinates[0])  # 10

# Unpacking tuples
x, y = coordinates
print(x, y)  # 10 20

Sets

# Sets contain unique items
numbers = {1, 2, 3, 3, 4}
print(numbers)  # {1, 2, 3, 4}

# Set operations
numbers.add(5)
numbers.remove(1)

File Input and Output

Reading Files

# Simple read
with open("data.txt", "r") as file:
    content = file.read()
    print(content)

# Read line by line
with open("data.txt", "r") as file:
    for line in file:
        print(line.strip())

Writing Files

# Write new file (creates if doesn't exist, overwrites if does)
with open("output.txt", "w") as file:
    file.write("Hello, Python!\n")
    file.write("This is line 2")

# Append to file
with open("output.txt", "a") as file:
    file.write("\nThis is a new line")

Working with CSV Files

import csv

# Reading CSV
with open("data.csv", "r") as file:
    reader = csv.reader(file)
    for row in reader:
        print(row)

# Writing CSV
with open("output.csv", "w", newline="") as file:
    writer = csv.writer(file)
    writer.writerow(["Name", "Age"])
    writer.writerow(["Alice", 25])

Error Handling Basics

Try-Except Blocks

# Basic error handling
try:
    number = int(input("Enter a number: "))
    result = 10 / number
    print(result)
except ValueError:
    print("That's not a valid number!")
except ZeroDivisionError:
    print("Cannot divide by zero!")

Finally Block

try:
    file = open("data.txt", "r")
    content = file.read()
except FileNotFoundError:
    print("File not found!")
finally:
    # This always runs
    if 'file' in locals():
        file.close()

Common Errors to Watch For

# NameError: Using undefined variable
# print(undefined_var)  # ERROR

# IndexError: Accessing list item that doesn't exist
# my_list = [1, 2, 3]
# print(my_list[10])  # ERROR

# KeyError: Accessing non-existent dictionary key
# person = {"name": "Alice"}
# print(person["age"])  # ERROR

Learning Tips and Best Practices

Best Practices for Beginners

  1. Write Code Regularly
    • Practice every day, even 15 minutes helps
    • Repetition builds muscle memory
  2. Read Other People's Code
    • Learn different approaches
    • See best practices in action
  3. Use Comments Wisely 
    # Calculate total price with tax
total = price * (1 + tax_rate)
  4. Test Your Code 
    # Always test with different inputs

    return a + b

  5. Keep It Simple (KISS)
    • Don't over-complicate solutions
    • Simple code is easier to debug

Practice Projects for Beginners

Project 1: Temperature Converter

def celsius_to_fahrenheit(celsius):
    return (celsius * 9/5) + 32

temp = float(input("Enter temperature in Celsius: "))
result = celsius_to_fahrenheit(temp)
print(f"{temp}°C = {result}°F")

Project 2: Number Guessing Game

import random

secret = random.randint(1, 100)
attempts = 0

while True:
    guess = int(input("Guess a number (1-100): "))
    attempts += 1
    
    if guess == secret:
        print(f"You won in {attempts} attempts!")
        break
    elif guess < secret:
        print("Too low, try higher")
    else:
        print("Too high, try lower")

Project 3: Simple To-Do List

tasks = []

while True:
    print("\n1. Add task\n2. View tasks\n3. Remove task\n4. Exit")
    choice = input("Choose: ")
    
    if choice == "1":
        task = input("Enter task: ")
        tasks.append(task)
    elif choice == "2":
        for i, task in enumerate(tasks):
            print(f"{i+1}. {task}")
    elif choice == "3":
        num = int(input("Task number to remove: "))
        if 0 < num <= len(tasks):
            tasks.pop(num-1)
    elif choice == "4":
        break
