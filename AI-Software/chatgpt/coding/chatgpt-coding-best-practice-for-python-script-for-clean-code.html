Introduction

Writing clean code is not just about making it work—it's about creating code that is readable, maintainable, and efficient. This guide explores ChatGPT-assisted techniques for writing clean, professional Python code that scales well and is easy for others to understand.

Code Formatting and Style

PEP 8 Guidelines

PEP 8 is Python's official style guide. Following it ensures consistency:

# Good: PEP 8 compliant
def calculate_total_price(items, tax_rate):
    """Calculate total price with tax."""
    subtotal = sum(item['price'] for item in items)
    total = subtotal * (1 + tax_rate)
    return total

# Bad: Not following conventions
def calc(items,tax):
    subtotal=sum([item['price'] for item in items])
    total=subtotal*(1+tax)
    return total

Line Length and Breaks

  • Keep lines to 79 characters (or 99 for flexibility)
  • Break long lines logically
# Good: Breaking at logical points
result = function_one(arg1, arg2) + \
         function_two(arg3, arg4)

# Also good: Using parentheses
result = (function_one(arg1, arg2) + 
          function_two(arg3, arg4))

Whitespace Usage

# Good: Proper spacing
if x > 5 and y < 10:
    result = {'key': 'value'}
    items = [1, 2, 3, 4, 5]

# Bad: Inconsistent spacing
if x>5and y<10:
    result = { 'key' : 'value' }
    items = [1,2,3,4,5]

Functions and Methods

Function Design Principles

Single Responsibility Principle:

# Good: Each function does one thing
def validate_user_input(user_input):
    return len(user_input) > 0 and '@' in user_input

def save_user_to_database(user_data):
    # Save logic here
    pass

# Bad: Multiple responsibilities
def process_user(user_input):
    if not user_input or '@' not in user_input:
        print("Invalid input")
        return False
    # Save to database
    # Send email
    # Log activity
    pass

Function Length

# Good: Focused, short function
def calculate_discount(price, customer_type):
    """Calculate discount based on customer type."""
    discounts = {
        'gold': 0.20,
        'silver': 0.10,
        'bronze': 0.05
    }
    discount_rate = discounts.get(customer_type, 0)
    return price * (1 - discount_rate)

# Bad: Too long with mixed concerns
def process_order(order_data, customer_type, apply_discount, 
                  send_email, update_inventory):
    # Hundreds of lines of code
    pass

Inheritance and Composition

# Good: Using composition over inheritance
class Logger:
    def log(self, message):
        print(f"[LOG] {message}")

class UserService:
    def __init__(self, logger, repository):
        self.logger = logger
        self.repository = repository
    
    def create_user(self, user_data):
        self.logger.log("Creating new user")
        return self.repository.create_user(user_data)

# Bad: Over-using inheritance
class UserService(Logger, Repository, Database):
    pass

Error Handling

Specific Exception Handling

# Good: Handle specific exceptions
def read_json_file(file_path):
    try:
        with open(file_path, 'r') as f:
            return json.load(f)
    except FileNotFoundError:
        print(f"File not found: {file_path}")
        return None
    except json.JSONDecodeError:
        print(f"Invalid JSON in: {file_path}")
        return None
    except Exception as e:
        print(f"Unexpected error: {e}")
        raise

# Bad: Catching all exceptions
def read_json_file(file_path):
    try:
        with open(file_path, 'r') as f:
            return json.load(f)
    except:
        return None

Custom Exceptions

# Good: Custom exceptions for clarity
class InsufficientFundsError(Exception):
    """Raised when account doesn't have enough balance."""
    pass

class InvalidTransactionError(Exception):
    """Raised when transaction parameters are invalid."""
    pass

def withdraw(account, amount):
    if amount <= 0:
        raise InvalidTransactionError("Amount must be positive")
    if account.balance < amount:
        raise InsufficientFundsError("Not enough balance")
    account.balance -= amount

def calculate_compound_interest(principal, rate, years):
    # Using the compound interest formula: A = P(1 + r/n)^(nt)
    # where n = 12 (compounded monthly)
    return principal * (1 + rate / 12) ** (years * 12)

def calculate_compound_interest(principal, rate, years):

Performance Optimization

Efficient Data Structures

# Good: Using appropriate data structures
def find_user_quickly(users_list, user_id):
    # Convert to dict for O(1) lookup
    users_dict = {user['id']: user for user in users_list}
    return users_dict.get(user_id)

# Bad: Linear search each time
def find_user_slowly(users_list, user_id):
    for user in users_list:
        if user['id'] == user_id:
            return user
    return None

List Comprehensions

# Good: List comprehension (faster and cleaner)
squared = [x ** 2 for x in range(1000) if x % 2 == 0]

# Acceptable: Traditional loop
squared = []
for x in range(1000):
    if x % 2 == 0:
        squared.append(x ** 2)

Security Considerations

Input Validation

# Good: Validate all inputs
def create_user(user_data):
    if not isinstance(user_data, dict):
        raise TypeError("user_data must be a dictionary")
    if 'email' not in user_data:
        raise ValueError("email is required")
    if not validate_email(user_data['email']):
        raise ValueError("Invalid email format")
    
    # Proceed with creation
    return User.create(user_data)

# Bad: No validation
def create_user(user_data):
    return User.create(user_data)

Secure Password Handling

# Good: Hash passwords with salt
import bcrypt

def hash_password(password):
    return bcrypt.hashpw(password.encode(), bcrypt.gensalt())

def verify_password(password, hashed):
    return bcrypt.checkpw(password.encode(), hashed)

# Bad: Storing plain text passwords
def create_user(username, password):
    user = User(username=username, password=password)
    user.save()

Refactoring Techniques

Reducing Complexity

# Bad: High cyclomatic complexity
def validate_user(user):
    if user.age < 18:
        return False
    elif user.age > 120:
        return False
    elif not user.email:
        return False
    elif not user.name:
        return False
    elif len(user.password) < 8:
        return False
    return True

# Good: Extract validation logic
def validate_user(user):
    """Validate user data."""
    validators = [
        is_valid_age,
        has_email,
        has_name,
        has_strong_password
    ]
    return all(validator(user) for validator in validators)

def is_valid_age(user):
    return 18 <= user.age <= 120
